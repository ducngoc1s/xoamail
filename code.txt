function main() {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var lastRow = sheet.getLastRow();
    var userColumn = 1;

    for (var i = 1; i <= lastRow; i++) {
        var userKey = sheet.getRange(i, userColumn).getValue();
        if (userKey) {
            // Split the string by '|'
            var parts = userKey.split('|');
            // Take the first part
            var userKeyBeforePipe = parts[0];
            xoaUser(userKeyBeforePipe, i);
        }
    }
}


function xoaNguoiDung(khoaNguoiDung, chiSoHang) {
    var bang = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    try {
        AdminDirectory.Users.remove(khoaNguoiDung);
        bang.getRange(chiSoHang, 2).setValue('Xóa Thành Công').setBackground('#1E90FF');
        Logger.log('Người dùng đã được xóa thành công: ' + khoaNguoiDung);
    } catch (e) {
        bang.getRange(chiSoHang, 2).setValue('Lỗi Khi Xóa').setBackground('#FF0000');
        Logger.log('Lỗi khi xóa người dùng ' + khoaNguoiDung + ': ' + e.toString());
    }
}
